Fonction count_word(word) : int 
    taille_word = longeur(word)
    retourner taille_word

i = 0

    word = ""
    Demander word

    si word != "STOP":
        taille = count_word(word)
        Ajouter taille à word_stock
        i += taille

(
    sinon si word == "STOP":
        i -= 4
)

    Fin si
Fin while

Afficher "Le total est de: ", i
